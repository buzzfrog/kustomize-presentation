# REPLACEMENTS #
- copy configmap.yaml to demo-space

- add to kustomization.yaml

----------
resources:
  - deploy.yaml
  - configmap.yaml

replacements:
- source:
    kind: Configmap
    name: active-user-id
    fieldPath: data.userid
  targets:
  - select:
      kind: Deployment
      name: app1
    fieldPaths:
    - spec.template.spec.containers.[name=app1-busybox].env.[name=USER_ID].value
----------
